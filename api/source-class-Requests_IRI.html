 239:     /**
 240:      * Create a new IRI object by resolving a relative IRI

 250:         if (!($relative instanceof Requests_IRI))

 336:      * Parse an IRI into scheme/authority/path/query/fragment segments
 337:      *
 338:      * @param string $iri
 339:      * @return array
 340:      */
 341:     private function parse_iri($iri)
 342:     {
 343:         $iri = trim($iri, "\x20\x09\x0A\x0C\x0D");
 344:         if (preg_match('/^((?P<scheme>[^:\/?#]+):)?(\/\/(?P<authority>[^\/?#]*))?(?P<path>[^?#]*)(\?(?P<query>[^#]*))?(#(?P<fragment>.*))?$/', $iri, $match))
 345:         {
 346:             if ($match[1] === '')
 347:             {
 348:                 $match['scheme'] = null;
 349:             }
 350:             if (!isset($match[3]) || $match[3] === '')
 351:             {
 352:                 $match['authority'] = null;
 353:             }
 354:             if (!isset($match[5]))
 355:             {
 356:                 $match['path'] = '';
 357:             }
 358:             if (!isset($match[6]) || $match[6] === '')
 359:             {
 360:                 $match['query'] = null;
 361:             }
 362:             if (!isset($match[8]) || $match[8] === '')
 363:             {
 364:                 $match['fragment'] = null;
 365:             }
 366:             return $match;
 367:         }
 368:         else
 369:         {
 370:             trigger_error('This should never happen', E_USER_ERROR);
 371:             die;
 372:         }
 373:     }

 375:     /**
 376:      * Remove dot segments from a path
 377:      *
 378:      * @param string $input
 379:      * @return string
 380:      */
 381:     private function remove_dot_segments($input)
 382:     {
 383:         $output = '';
 384:         while (strpos($input, './') !== false || strpos($input, '/.') !== false || $input === '.' || $input === '..')
 385:         {
 386:             // A: If the input buffer begins with a prefix of "../" or "./", then remove that prefix from the input buffer; otherwise,
 387:             if (strpos($input, '../') === 0)
 388:             {
 389:                 $input = substr($input, 3);
 390:             }
 391:             elseif (strpos($input, './') === 0)
 392:             {
 393:                 $input = substr($input, 2);
 394:             }
 395:             // B: if the input buffer begins with a prefix of "/./" or "/.", where "." is a complete path segment, then replace that prefix with "/" in the input buffer; otherwise,
 396:             elseif (strpos($input, '/./') === 0)
 397:             {
 398:                 $input = substr($input, 2);
 399:             }
 400:             elseif ($input === '/.')
 401:             {
 402:                 $input = '/';
 403:             }
 404:             // C: if the input buffer begins with a prefix of "/../" or "/..", where ".." is a complete path segment, then replace that prefix with "/" in the input buffer and remove the last segment and its preceding "/" (if any) from the output buffer; otherwise,
 405:             elseif (strpos($input, '/../') === 0)
 406:             {
 407:                 $input = substr($input, 3);
 408:                 $output = substr_replace($output, '', strrpos($output, '/'));
 409:             }
 410:             elseif ($input === '/..')
 411:             {
 412:                 $input = '/';
 413:                 $output = substr_replace($output, '', strrpos($output, '/'));
 414:             }
 415:             // D: if the input buffer consists only of "." or "..", then remove that from the input buffer; otherwise,
 416:             elseif ($input === '.' || $input === '..')
 417:             {
 418:                 $input = '';
 419:             }
 420:             // E: move the first path segment in the input buffer to the end of the output buffer, including the initial "/" character (if any) and any subsequent characters up to, but not including, the next "/" character or the end of the input buffer
 421:             elseif (($pos = strpos($input, '/', 1)) !== false)
 422:             {
 423:                 $output .= substr($input, 0, $pos);
 424:                 $input = substr_replace($input, '', 0, $pos);
 425:             }
 426:             else
 427:             {
 428:                 $output .= $input;
 429:                 $input = '';
 430:             }
 431:         }
 432:         return $output . $input;
 433:     }

 435:     /**
 436:      * Replace invalid character with percent encoding
 437:      *
 438:      * @param string $string Input string
 439:      * @param string $extra_chars Valid characters not in iunreserved or
 440:      *                            iprivate (this is ASCII-only)
 441:      * @param bool $iprivate Allow iprivate
 442:      * @return string
 443:      */
 444:     private function replace_invalid_with_pct_encoding($string, $extra_chars, $iprivate = false)
 445:     {
 446:         // Normalize as many pct-encoded sections as possible
 447:         $string = preg_replace_callback('/(?:%[A-Fa-f0-9]{2})+/', array(&$this, 'remove_iunreserved_percent_encoded'), $string);
 448: 
 449:         // Replace invalid percent characters
 450:         $string = preg_replace('/%(?![A-Fa-f0-9]{2})/', '%25', $string);
 451: 
 452:         // Add unreserved and % to $extra_chars (the latter is safe because all
 453:         // pct-encoded sections are now valid).
 454:         $extra_chars .= 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-._~%';
 455: 
 456:         // Now replace any bytes that aren't allowed with their pct-encoded versions
 457:         $position = 0;
 458:         $strlen = strlen($string);
 459:         while (($position += strspn($string, $extra_chars, $position)) < $strlen)
 460:         {
 461:             $value = ord($string[$position]);
 462: 
 463:             // Start position
 464:             $start = $position;
 465: 
 466:             // By default we are valid
 467:             $valid = true;
 468: 
 469:             // No one byte sequences are valid due to the while.
 470:             // Two byte sequence:
 471:             if (($value & 0xE0) === 0xC0)
 472:             {
 473:                 $character = ($value & 0x1F) << 6;
 474:                 $length = 2;
 475:                 $remaining = 1;
 476:             }
 477:             // Three byte sequence:
 478:             elseif (($value & 0xF0) === 0xE0)
 479:             {
 480:                 $character = ($value & 0x0F) << 12;
 481:                 $length = 3;
 482:                 $remaining = 2;
 483:             }
 484:             // Four byte sequence:
 485:             elseif (($value & 0xF8) === 0xF0)
 486:             {
 487:                 $character = ($value & 0x07) << 18;
 488:                 $length = 4;
 489:                 $remaining = 3;
 490:             }
 491:             // Invalid byte:
 492:             else
 493:             {
 494:                 $valid = false;
 495:                 $length = 1;
 496:                 $remaining = 0;
 497:             }
 498: 
 499:             if ($remaining)
 500:             {
 501:                 if ($position + $length <= $strlen)
 502:                 {
 503:                     for ($position++; $remaining; $position++)
 504:                     {
 505:                         $value = ord($string[$position]);
 506: 
 507:                         // Check that the byte is valid, then add it to the character:
 508:                         if (($value & 0xC0) === 0x80)
 509:                         {
 510:                             $character |= ($value & 0x3F) << (--$remaining * 6);
 511:                         }
 512:                         // If it is invalid, count the sequence as invalid and reprocess the current byte:
 513:                         else
 514:                         {
 515:                             $valid = false;
 516:                             $position--;
 517:                             break;
 518:                         }
 519:                     }
 520:                 }
 521:                 else
 522:                 {
 523:                     $position = $strlen - 1;
 524:                     $valid = false;
 525:                 }
 526:             }
 527: 
 528:             // Percent encode anything invalid or not in ucschar
 529:             if (
 530:                 // Invalid sequences
 531:                 !$valid
 532:                 // Non-shortest form sequences are invalid
 533:                 || $length > 1 && $character <= 0x7F
 534:                 || $length > 2 && $character <= 0x7FF
 535:                 || $length > 3 && $character <= 0xFFFF
 536:                 // Outside of range of ucschar codepoints
 537:                 // Noncharacters
 538:                 || ($character & 0xFFFE) === 0xFFFE
 539:                 || $character >= 0xFDD0 && $character <= 0xFDEF
 540:                 || (
 541:                     // Everything else not in ucschar
 542:                        $character > 0xD7FF && $character < 0xF900
 543:                     || $character < 0xA0
 544:                     || $character > 0xEFFFD
 545:                 )
 546:                 && (
 547:                     // Everything not in iprivate, if it applies
 548:                        !$iprivate
 549:                     || $character < 0xE000
 550:                     || $character > 0x10FFFD
 551:                 )
 552:             )
 553:             {
 554:                 // If we were a character, pretend we weren't, but rather an error.
 555:                 if ($valid)
 556:                     $position--;
 557: 
 558:                 for ($j = $start; $j <= $position; $j++)
 559:                 {
 560:                     $string = substr_replace($string, sprintf('%%%02X', ord($string[$j])), $j, 1);
 561:                     $j += 2;
 562:                     $position += 2;
 563:                     $strlen += 2;
 564:                 }
 565:             }
 566:         }
 567: 
 568:         return $string;
 569:     }

 571:     /**
 572:      * Callback function for preg_replace_callback.
 573:      *
 574:      * Removes sequences of percent encoded bytes that represent UTF-8
 575:      * encoded characters in iunreserved
 576:      *
 577:      * @param array $match PCRE match
 578:      * @return string Replacement
 579:      */
 580:     private function remove_iunreserved_percent_encoded($match)
 581:     {
 582:         // As we just have valid percent encoded sequences we can just explode
 583:         // and ignore the first member of the returned array (an empty string).
 584:         $bytes = explode('%', $match[0]);
 585: 
 586:         // Initialize the new string (this is what will be returned) and that
 587:         // there are no bytes remaining in the current sequence (unsurprising
 588:         // at the first byte!).
 589:         $string = '';
 590:         $remaining = 0;
 591: 
 592:         // Loop over each and every byte, and set $value to its value
 593:         for ($i = 1, $len = count($bytes); $i < $len; $i++)
 594:         {
 595:             $value = hexdec($bytes[$i]);
 596: 
 597:             // If we're the first byte of sequence:
 598:             if (!$remaining)
 599:             {
 600:                 // Start position
 601:                 $start = $i;
 602: 
 603:                 // By default we are valid
 604:                 $valid = true;
 605: 
 606:                 // One byte sequence:
 607:                 if ($value <= 0x7F)
 608:                 {
 609:                     $character = $value;
 610:                     $length = 1;
 611:                 }
 612:                 // Two byte sequence:
 613:                 elseif (($value & 0xE0) === 0xC0)
 614:                 {
 615:                     $character = ($value & 0x1F) << 6;
 616:                     $length = 2;
 617:                     $remaining = 1;
 618:                 }
 619:                 // Three byte sequence:
 620:                 elseif (($value & 0xF0) === 0xE0)
 621:                 {
 622:                     $character = ($value & 0x0F) << 12;
 623:                     $length = 3;
 624:                     $remaining = 2;
 625:                 }
 626:                 // Four byte sequence:
 627:                 elseif (($value & 0xF8) === 0xF0)
 628:                 {
 629:                     $character = ($value & 0x07) << 18;
 630:                     $length = 4;
 631:                     $remaining = 3;
 632:                 }
 633:                 // Invalid byte:
 634:                 else
 635:                 {
 636:                     $valid = false;
 637:                     $remaining = 0;
 638:                 }
 639:             }
 640:             // Continuation byte:
 641:             else
 642:             {
 643:                 // Check that the byte is valid, then add it to the character:
 644:                 if (($value & 0xC0) === 0x80)
 645:                 {
 646:                     $remaining--;
 647:                     $character |= ($value & 0x3F) << ($remaining * 6);
 648:                 }
 649:                 // If it is invalid, count the sequence as invalid and reprocess the current byte as the start of a sequence:
 650:                 else
 651:                 {
 652:                     $valid = false;
 653:                     $remaining = 0;
 654:                     $i--;
 655:                 }
 656:             }
 657: 
 658:             // If we've reached the end of the current byte sequence, append it to Unicode::$data
 659:             if (!$remaining)
 660:             {
 661:                 // Percent encode anything invalid or not in iunreserved
 662:                 if (
 663:                     // Invalid sequences
 664:                     !$valid
 665:                     // Non-shortest form sequences are invalid
 666:                     || $length > 1 && $character <= 0x7F
 667:                     || $length > 2 && $character <= 0x7FF
 668:                     || $length > 3 && $character <= 0xFFFF
 669:                     // Outside of range of iunreserved codepoints
 670:                     || $character < 0x2D
 671:                     || $character > 0xEFFFD
 672:                     // Noncharacters
 673:                     || ($character & 0xFFFE) === 0xFFFE
 674:                     || $character >= 0xFDD0 && $character <= 0xFDEF
 675:                     // Everything else not in iunreserved (this is all BMP)
 676:                     || $character === 0x2F
 677:                     || $character > 0x39 && $character < 0x41
 678:                     || $character > 0x5A && $character < 0x61
 679:                     || $character > 0x7A && $character < 0x7E
 680:                     || $character > 0x7E && $character < 0xA0
 681:                     || $character > 0xD7FF && $character < 0xF900
 682:                 )
 683:                 {
 684:                     for ($j = $start; $j <= $i; $j++)
 685:                     {
 686:                         $string .= '%' . strtoupper($bytes[$j]);
 687:                     }
 688:                 }
 689:                 else
 690:                 {
 691:                     for ($j = $start; $j <= $i; $j++)
 692:                     {
 693:                         $string .= chr(hexdec($bytes[$j]));
 694:                     }
